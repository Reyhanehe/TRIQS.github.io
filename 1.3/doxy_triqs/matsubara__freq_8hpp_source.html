157  long L;
158  int s = 1;
159  h5_read(gr, "domain", dom);
160  h5_read(gr, "size", L);
161  // backward compatibility : older file do not have this flags, default is true.
162  if (gr.has_key("positive_freq_only")) h5_read(gr, "positive_freq_only", s);
163  if (gr.has_key("start_at_0")) h5_read(gr, "start_at_0", s);
164  m = gf_mesh{std::move(dom), L, (s==1)};
165  }
