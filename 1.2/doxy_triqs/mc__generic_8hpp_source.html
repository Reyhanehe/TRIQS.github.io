94  bool start(MCSignType sign_init, std::function<bool ()> const & stop_callback) {
95  assert(stop_callback);
96  Timer.start();
97  sign = sign_init; done_percent = 0; nmeasures = 0;
98  sum_sign = 0;
99  bool stop_it=false, finished = false;
100  uint64_t NCycles_tot = NCycles+ NWarmIterations;
101  report << std::endl << std::flush;
102  for (NC =0; !stop_it; ++NC) {
103  for (uint64_t k=1; (k<=Length_MC_Cycle); k++) { MCStepType::do_it(AllMoves,RandomGenerator,sign); }
104  if (after_cycle_duty) {after_cycle_duty();}
105  if (thermalized()) {
106  nmeasures++;
107  sum_sign += sign;
108  for (auto &x : AllMeasuresAux) x();
109  AllMeasures.accumulate(sign);
110  }
111  // recompute fraction done
112  uint64_t dp = uint64_t(floor( ( NC*100.0) / (NCycles_tot-1)));
113  if (dp>done_percent) { done_percent=dp; report << done_percent; report<<"%; "; report <<std::flush; }
114  finished = ( (NC >= NCycles_tot -1) || converged () );
115  stop_it = (stop_callback() || finished);
116  }
117  report << std::endl << std::endl << std::flush;
118  Timer.stop();
119  return finished;
120 
121  }

124  void collect_results(boost::mpi::communicator const & c) {
125 
126  uint64_t nmeasures_tot;
127  MCSignType sum_sign_tot;
128  boost::mpi::reduce(c, nmeasures, nmeasures_tot, std::plus<uint64_t>(), 0);
129  boost::mpi::reduce(c, sum_sign, sum_sign_tot, std::plus<MCSignType>(), 0);
130 
131  report(3) << "[Node "<<c.rank()<<"] Acceptance rate for all moves:\n" << AllMoves.get_statistics(c);
132  report(3) << "[Node "<<c.rank()<<"] Simulation lasted: " << double(Timer) << " seconds" << std::endl;
133  report(3) << "[Node "<<c.rank()<<"] Number of measures: " << nmeasures << std::endl;
134  report(3) << "[Node "<<c.rank()<<"] Average sign: " << sum_sign / double(nmeasures) << std::endl << std::endl << std::flush;
135 
136  if (c.rank()==0) {
137  sign_av = sum_sign_tot / double(nmeasures_tot);
138  report(2) << "Total number of measures: " << nmeasures_tot << std::endl;
139  report(2) << "Average sign: " << sign_av << std::endl << std::endl << std::flush;
140  }
141  boost::mpi::broadcast(c, sign_av, 0);
142  AllMeasures.collect_results(c);
143 
144  }
