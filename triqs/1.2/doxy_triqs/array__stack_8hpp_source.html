44  template <typename T, int R> class array_stack_impl {
45  static const size_t dim = R;
46  static const bool base_is_array = dim > 0;
47  size_t bufsize_, step, _size;
48  static const bool T_is_complex = triqs::is_complex<T>::value;
49  static const unsigned int RANK = dim + 1 + (T_is_complex ? 1 : 0);
50  utility::mini_vector<hsize_t, RANK> dims, offset, maxdims, dim_chunk, buffer_dim, zero;
51  h5::dataset d_set;
52  array<T, dim + 1> buffer;
53 
54  public:
55  array_stack_impl(h5::group g, std::string const &name, mini_vector<size_t, dim> const &base_element_shape, size_t bufsize) {
56  mini_vector<hsize_t, RANK> dim_chunk;
57  bufsize_ = bufsize;
58  step = 0;
59  _size = 0;
60  for (size_t i = 1; i <= dim; ++i) {
61  dims[i] = base_element_shape[i - 1];
62  }
63  if (T_is_complex) {
64  dims[RANK - 1] = 2;
65  }
66  maxdims = dims;
67  buffer_dim = dims;
68  dim_chunk = dims;
69  dims[0] = 0;
70  maxdims[0] = H5S_UNLIMITED;
71  dim_chunk[0] = 1;
72  buffer_dim[0] = bufsize_;
73  mini_vector<size_t, dim + 1> s;
74  for (size_t i = 0; i <= dim; ++i) {
75  s[i] = buffer_dim[i];
76  }
77  buffer.resize(s);
78  h5::dataspace mspace1 = H5Screate_simple(RANK, dims.ptr(), maxdims.ptr());
79  h5::proplist cparms = H5Pcreate(H5P_DATASET_CREATE);
80  //auto err = H5Pset_chunk(cparms, RANK, dim_chunk.ptr());
81  H5Pset_chunk(cparms, RANK, dim_chunk.ptr());
82  d_set = g.create_dataset(name, h5::native_type_from_C(T()), mspace1, cparms);
83  if (triqs::is_complex<T>::value) h5::write_string_attribute(d_set, "__complex__", "1");
84  }
85 
87  ~array_stack_impl() { flush(); }

101  void operator++() {
102  ++step;
103  ++_size;
104  if (step == bufsize_) flush();
105  }
106 
108  void flush() {
109  save_buffer();
110  step = 0;
111  }
112 
117  template <class AType> void operator<<(AType const &A) {
118  (*this)() = A;
119  ++(*this);
120  }
121 
123  size_t size() const { return _size; }
124 
125  private:
126  void save_buffer() {
127  if (step == 0) return;
128  dims[0] += step;
129  buffer_dim[0] = step;
130 
131  herr_t err= H5Dset_extent(d_set,dims.ptr()); // resize the data_space
132 
133  h5::dataspace fspace1 = H5Dget_space(d_set);
134  h5::dataspace mspace = h5_impl::data_space(buffer);
135 
136  err = H5Sselect_hyperslab(fspace1, H5S_SELECT_SET, offset.ptr(), NULL, buffer_dim.ptr(), NULL);
137  if (err < 0) TRIQS_RUNTIME_ERROR << "Cannot set hyperslab";
138  err = H5Sselect_hyperslab(mspace, H5S_SELECT_SET, zero.ptr(), NULL, buffer_dim.ptr(), NULL);
139  if (err < 0) TRIQS_RUNTIME_ERROR << "Cannot set hyperslab";
140 
141  err = H5Dwrite(d_set, h5::data_type_memory<T>(), mspace, fspace1, H5P_DEFAULT, h5_impl::__get_array_data_ptr(buffer));
142  if (err < 0) TRIQS_RUNTIME_ERROR << "Error writing the array_stack buffer";
143  offset[0] += step;
144  }
145  };
